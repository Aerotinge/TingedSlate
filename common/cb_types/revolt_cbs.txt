# The CB used by "Peasant revolter" characters
peasant_revolt = {
	name = CB_NAME_PEASANT_CONQUEST
	war_name = WAR_NAME_PEASANT_CONQUEST
	sprite = 16
	truce_days = 365

	hostile_against_others = yes
	check_all_titles = yes # if permanent, setting this to true will check against all of someones titles, including vassal held titles
	attacker_can_call_allies = no
	defender_can_call_allies = no
	can_ask_to_join_war = no
	is_independence = yes

	can_use = {
		# Attacker
		ROOT = {
			primary_title = { temporary = yes } # Revolter or adventurer trigger
			war = no
			NOT = { is_vassal_or_below_of = FROM }
		}
	}

	can_use_title = {
		real_tier = COUNT
	}

	is_valid_title = {
		holder_scope = {
			OR = {
				character = FROM
				is_vassal_or_below_of = FROM
			}
		}
	}

	on_add = {
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}

	on_success_title = {
		usurp_title_plus_barony_if_unlanded = {
			target = attacker
			type = revolt
		}

		any_de_jure_vassal_title = { # take all baronies too
			show_scope_change = no

			limit = {
				has_holder = yes

				holder_scope = {
					OR = {
						character = defender
						is_vassal_or_below_of = defender

						AND = {
							is_vassal_or_below_of = attacker

							NOT = { trait = peasant_leader }
						}
					}
				}
			}

			usurp_title = attacker
		}

		attacker = {
			show_scope_change = no

			set_defacto_liege = attacker
			if = { # Goes Peasant Republic if available
				limit = {
					capital_scope = {
						any_province_holding = {
							OR = {
								holding_type = city
								holding_type = castle
							}
							holder = attacker
						}
					}
				}

				capital_scope = {
					random_province_holding = {
						limit = {
							is_capital = no
							OR = {
								holding_type = city
								holding_type = castle
							}
							holder = attacker
						}
						preferred_limit = {
							holding_type = city
						}

						make_capital_holding = yes
					}
				}
				any_demesne_title = { # Required for set_government_type below
					limit = {
						rebel = yes
					}

					unsafe_destroy_landed_title = THIS
				}
				set_government_type = republic_government
			}
			recalc_succession = yes
		}

		defender = {
			show_scope_change = no

			prestige = -100
		}

		hidden_effect = {
			check_if_reconquista_finished_effect = yes

			defender = {
				landless_pope_effect = yes
			}
		}
	}

	on_fail_title = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 10

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 20

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}

				add_character_modifier = {
					name = crushed_revolt
					years = 10
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}

				}

				any_demesne_province = {
					random_province_holding = {
						limit = {
							is_capital = no
							OR = {
								holding_type = temple
								holding_type = castle
							}
							holder = attacker
						}
						preferred_limit = {
							holding_type = temple
						}

						make_capital_holding = yes
					}
				}
				any_demesne_title = { # Required for set_government_type below
					limit = {
						rebel = yes
					}

					unsafe_destroy_landed_title = THIS
				}
				set_government_type = theocracy_government
			}
			recalc_succession = yes
		}

		hidden_effect = {
			check_if_reconquista_finished_effect = yes

			defender = {
				landless_pope_effect = yes
			}
		}
	}

	on_fail = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = heretic_revolter

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 50
			piety = 100

			religion_authority = {
				name = won_heretic_revolt
			}

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}
			}
		}
	}

	on_reverse_demand = {

			imprison = defender

		has_holder = no
	}

		# Attacker
		ROOT = {
			NOT = { culture = FROM }
		}

	is_valid_title = {
		has_holder = no
	}

		fire_haruspicy_event_effect = yes
		anger_china_effect = yes

		attacker = {
			show_scope_change = no

		}

		defender = {
			show_scope_change = no

		}

					type = revolt
				}

				gain_all_occupied_titles = {
					who = defender
					type = revolt
				}

					limit = {
						num_of_count_titles >= 5

						PREV = {
							OR = {
								has_holder = no

			}
		}
	}

	on_fail_title = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = nationalist_rebel

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}

				remove_trait = inspiring_leader
			}
		}

		defender = {
			show_scope_change = no

			prestige = 100

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = nationalist_rebel

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}

				remove_trait = inspiring_leader
			}
		}
