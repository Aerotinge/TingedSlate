########################################################
# MINOR TITLES
# Characters can have any number of these
########################################################
#	grant_limit			Max number of holders
#	realm_in_name		Append the name of the realm after the title (Queen [of Sweden])
#	allowed_to_hold		Conditions for the character to hold the title
#	allowed_to_grant	Conditions for a ruler to grant the title
#	gain_effect			Fires when the title is granted
#	lose_effect			Fires when a title is lost through most causes
#	retire_effect		Fires when a title is lost due to the character stepping down
#	death_effect		Fires when a title is lost due to the character dying
#	message				Should fire a message to the its owner when set
#	The other fields are all character modifiers

#	Scopes (except allowed_to_grant)
#	ROOT is potential holder of minor title
#	FROM is liege

#	ai_will_do is supported

title_pronoiar = {
	message = no

	dignity = 0.8

	allowed_to_grant = {
		always = no
	}

	allowed_to_hold = {
	}

	revoke_trigger = {
		always = no
	}

	gain_effect = {
	}

	lose_effect = {
		set_character_flag = lose_pronoiar_delayed
		character_event = {
			id = EMP.21112
			days = 1
		}
	}

	retire_effect = {
	}

	death_effect = {
	}
}

title_markgraf = {
	is_unique = yes
	message = yes

	dignity = 0.05
	grant_limit = 1

	opinion_effect = 5
	monthly_salary = 0.01
	monthly_prestige = 0.50

	allowed_to_grant = {
		always = no # Disable for now, require better history simulation
		higher_tier_than = DUKE
		government = feudal_government

		OR = {
			religion_group = christian
			religion_group = jewish_group
		}

		NOR = {
			has_landed_title = e_byzantium
			has_landed_title = e_roman_empire
			has_landed_title = e_nicaea
		}
	}

	allowed_to_hold = {
		any_demesne_province = {
			NOT = {
				any_neighbor_province = {
					same_realm = FROM
				}
			}
		}

		capital_scope = {
			duchy = {
				holder = FROM
			}
		}

		NOT = { higher_tier_than = DUKE }
		government = feudal_government
		is_adult = yes
		is_incapable = no

		custom_tooltip = {
			text = court_physician_tooltip_1

			trigger_if = {
				limit = { is_ruler = yes }

				primary_title = {
					temporary = no
				}
			}
		}
	}

	allow = { # Not sure this working or not
		real_tier = DUKE
		government = feudal_government
		persistent_event_target:march_title = { holder = ROOT }
	}

	revoke_trigger = {
	}

	gain_effect = {
		capital_scope = {
			duchy = {
				ROOT = {
					save_persistent_event_target = {
						name = march_title
						scope = PREV
					}
					grant_title_no_opinion = {
						target = PREV
					}
				}
			}
		}

		add_character_modifier = {
			name = hidden_markgraf_bonus
			duration = -1
			hidden = yes
		}
	}

	lose_effect = {
		persistent_event_target:march_title = {
			if = {
				limit = {
					always = yes
					holder = ROOT
				}

				FROM = {
					add_claim = PREV
				}
			}
		}

		clear_persistent_event_target = march_title
		remove_character_modifier = hidden_markgraf_bonus
	}

	retire_effect = {
	}

	death_effect = {
	}
}
