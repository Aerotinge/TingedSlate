namespace = EMP

# Pentarchy Scheme
# Every landed kingdom title had a de jure pentarchy, which can be "drifted" by a persistent_event_target:pentarch_de_jure_drift
# A persistent_event_target:pentarch_de_facto = { title = THIS } can be used to make the kingdom autocephalous
# A persistent_event_target:pentarch_de_facto = { title = b_hagiasophia/b_antioch/... } is used when a pentarchy see is vacant, and redirect it to a valid pentarchy see.
# set_de_jure_pentarch_title_effect setup pentarch of a title, will abort if a persistent_event_target:pentarch_de_facto exists

# Setup de jure jurisdiction of the patriarchate under the Pentarchy
character_event = {
	id = EMP.20510

	is_triggered_only = yes # on_startup
	hide_window = yes

	trigger = {
		is_multiplayer_host_character = yes
		is_save_game = no
	}

	immediate = {
		any_landed_title = {
			limit = { tier = KING }

			save_event_target_as = ecclesiastical_eparchy
			set_de_jure_pentarch_title_effect = yes
		}

		set_global_flag = de_jure_pentarch_init
	}
}

# Setup Pentarchy for orthodox only
character_event = {
	id = EMP.20511

	is_triggered_only = yes # on_startup
	hide_window = yes

	trigger = {
		is_multiplayer_host_character = yes
		is_save_game = no
	}

	immediate = {
		orthodox	= {}
		iconoclast	= { set_pentarchy = no } # Autocephalous until it takes over orthodox
		monothelite	= { set_pentarchy = no }
		paulician	= { set_pentarchy = no }
	}
}

# on_heresy_takeover, norm get pentarchy and disabled for old orthodoxy
# So the heretic one become autocephalous only
character_event = {
	id = EMP.20512

	is_triggered_only = yes # on_heresy_takeover
	hide_window = yes

	trigger = {
		FROMFROM = { has_pentarchy = yes }
		FROM = {
			OR = {
				religion = orthodox
				religion = iconoclast
				religion = monothelite
				religion = paulician
			}
		}
	}

	immediate = {
		FROMFROM	= { set_pentarchy = no }
		FROM		= { set_pentarchy = yes }
	}
}

character_event = {
	id = EMP.20520

	is_triggered_only = yes # on_new_holder(_inheritance/usurpation)
	hide_window = yes

	trigger = {
		NOR = {
			has_pentarchy = yes
			government = theocracy_government
		}
		FROM = {
			OR = {
				title = b_roma
				title = b_alexandria
				title = b_antioch
				title = b_jerusalem
				title = b_hagiasophia
				trigger_if = {
					limit = {
						location = {
							capital_holding = { title = b_roma }
						}
					}
					title = c_roma
				}
				trigger_else_if = {
					limit = {
						location = {
							capital_holding = { title = b_alexandria }
						}
					}
					title = c_alexandria
				}
				trigger_else_if = {
					limit = {
						location = {
							capital_holding = { title = b_antioch }
						}
					}
					title = c_antioch
				}
				trigger_else_if = {
					limit = {
						location = {
							capital_holding = { title = b_jerusalem }
						}
					}
					title = c_jerusalem
				}
				trigger_else_if = {
					limit = {
						location = {
							capital_holding = { title = b_hagiasophia }
						}
					}
					title = c_constantinopolis
				}
			}
		}
	}

	immediate = {
		any_landed_title = {
			limit = {
				tier = KING
				persistent_event_target:pentarch_de_facto = {
					OR = {
						title = FROM
						county = {
							title = FROM
						}
					}
				}
			}

			clear_persistent_event_target = pentarch_de_facto
			if = {
				limit = {
					monothelite = {
						NOT = { # Is not heresy
							parent_religion = {
								always = yes
							}
						}
					}
				}
				set_pentarch_title = b_antioch
				save_persistent_event_target = {
					name = pentarch_de_facto
					scope = b_antioch
				}
			}
			else = {
				set_pentarch_title = b_hagiasophia
				save_persistent_event_target = {
					name = pentarch_de_facto
					scope = b_hagiasophia
				}
			}
		}
	}
}

character_event = {
	id = EMP.20530

	is_triggered_only = yes # on_character_convert_religion
	hide_window = yes

	trigger = {
		OR = {
			FROM = { has_pentarchy = yes }
			has_pentarchy = yes
		}
		OR = {
			has_landed_title = b_roma
			has_landed_title = b_alexandria
			has_landed_title = b_antioch
			has_landed_title = b_jerusalem
			has_landed_title = b_hagiasophia
		}
	}

	immediate = {
		if = {
			limit = { has_pentarchy = yes }

			any_landed_title = {
				limit = {
					tier = KING
					has_de_jure_pentarchy_pretension = yes
					persistent_event_target:pentarch_de_facto = {
						tier = BARON
					}
				}

				log = "[This.GetName] reset pentarch to the default"
				clear_persistent_event_target = pentarch_de_facto
				save_event_target_as = ecclesiastical_eparchy
				set_de_jure_pentarch_title_effect = yes
			}
		}
		else = {
			any_landed_title = {
				limit = {
					tier = KING
					NOT = { has_de_jure_pentarchy_pretension = yes }
					NOT = {
						persistent_event_target:pentarch_de_facto = {
							always = yes
						}
					}
				}

				log = "[This.GetName] is under a fallback pentarch due to absence of pentarch see"
				if = {
					limit = {
						ROOT = { religion = monothelite }
					}
					set_pentarch_title = b_antioch
					save_persistent_event_target = {
						name = pentarch_de_facto
						scope = b_antioch
					}
				}
				else = {
					set_pentarch_title = b_hagiasophia
					save_persistent_event_target = {
						name = pentarch_de_facto
						scope = b_hagiasophia
					}
				}
			}
		}
	}
}

			has_landed_title = b_roma

			has_landed_title = b_antioch
			has_landed_title = b_jerusalem

			primary_title = {
				NOT = {
					persistent_event_target:pentarch_de_facto = {
						title = THIS
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {

		}
	}
