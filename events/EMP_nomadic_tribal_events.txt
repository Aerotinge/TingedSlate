character_event = {
    id = EMP.11002
    hide_window = yes
	lacks_dlc = "Horse Lords"
	
    is_triggered_only = yes

    trigger = {
        government = nomadic_tribal_government
    }

	immediate = {
		set_variable = { which = "empty_holdings_total" value = 0 } # empty_holdings_total cap at 35
		# calc empty holding in realm
		any_realm_province = {
			limit = {
				OR = {
					terrain = steppe
					terrain = plains
					terrain = desert
					terrain = farmlands
				}
				owner = {
					government = nomadic_tribal_government
				}
			}
			ROOT = {
				export_to_variable = {
					which = empty_holdings
					value = num_of_empty_holdings
					who = PREV
				}
				change_variable = {
					which = empty_holdings_total
					which = empty_holdings
					who = ROOT
				}
			}
		}
		ROOT = {
			# over demesne penalty
			export_to_variable = {
				which = over_prov_total
				value = over_max_demesne_size
			}
			multiply_variable = {
				which = over_prov_total
				value = 5
			}
			trigger_if = {
				limit = {
					total_years_played > 1
				}
				subtract_variable = {
					which = empty_holdings_total
					which = over_prov_total
				}
			}
			# Base coefficient
			divide_variable = {
				which = empty_holdings_total
				value = 2
			}
			# and modified by martial of ROOT
			export_to_variable = {
				which = character_martial
				value = martial
			}
			trigger_if = {
				limit = {
					check_variable = {
						which = character_martial
						value > 20
					}
				}
				set_variable = {
					which = character_martial
					value = 20
				}
			}
			change_variable = {
				which = character_martial
				value = -10
			}
			divide_variable = {
				which = character_martial
				value = 50
			}
			change_variable = {
				which = character_martial
				value = 1
			}
			multiply_variable = {
				which = empty_holdings_total
				which = character_martial
			}
			# Boundary check
			if = {
				limit = {
					check_variable = {
						which = empty_holdings_total
						value > 35
					}
				}
				set_variable = {
					which = empty_holdings_total
					value = 35
				}
			}
			else_if = {
				limit = {
					check_variable = {
						which = empty_holdings_total
						value < 1
					}
				}
				set_variable = {
					which = empty_holdings_total
					value = 1
				}
			}
			# cleanup
			set_variable = {
				which = empty_holdings
				value = 0
			}
			set_variable = {
				which = character_martial
				value = 0
			}
			set_variable = {
				which = over_prov_total
				value = 0
			}
		}
	}
}
